    def saisie(tab):
        taille = len(tab)
        print('Vous allez saisir', taille, 'valeurs')
        for i in range(taille):
            tab[i] = eval(input('\n\t tab['+str(i)+'] = ? '))

"""""""""
.. code-block:: python
    :linenos:

    def affichage(tab):
        taille = len(tab)
        print('Affichege de ', taille, 'valeurs ')
        for i in range(taille):
            print('tab[', i, '] = ', tab[i], end='; ')
